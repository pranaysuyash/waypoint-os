1. Dropzone.tsx — drag & drop with preview

'use client'
import { useCallback, useState } from 'react'

interface DropzoneProps {
  onFileSelect: (file: File) => void
  destination?: string
  accept?: string
}

export default function Dropzone({ onFileSelect, destination }: DropzoneProps) {
  const [isDragging, setIsDragging] = useState(false)
  const [fileName, setFileName] = useState(null)

  const handleDrop = useCallback((e: React.DragEvent) => {
    e.preventDefault()
    setIsDragging(false)
    const file = e.dataTransfer.files?.[0]
    if (file && (file.type === 'application/pdf' || file.type.startsWith('image/'))) {
      setFileName(file.name)
      onFileSelect(file)
    }
  }, [onFileSelect])

  return (
    <div
      onDragOver={(e) => { e.preventDefault(); setIsDragging(true) }}
      onDragLeave={() => setIsDragging(false)}
      onDrop={handleDrop}
      className={`relative border-2 border-dashed rounded-2xl p-10 text-center transition
        ${isDragging ? 'border-violet-500 bg-violet-500/10' : 'border-zinc-700 hover:border-zinc-600 bg-zinc-900/50'}`}
    >
      <input
        type="file"
        accept=".pdf,image/*"
        className="absolute inset-0 w-full h-full opacity-0 cursor-pointer"
        onChange={(e) => {
          const file = e.target.files?.[0]
          if (file) { setFileName(file.name); onFileSelect(file) }
        }}
      />
      <div className="space-y-3">
        <div className="w-12 h-12 mx-auto rounded-xl bg-zinc-800 flex items-center justify-center">
          <svg width="24" height="24" fill="none" stroke="currentColor" strokeWidth="2"><path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4M17 8l-5-5-5 5M12 3v12"/></svg>
        </div>
        <div>
          <p className="font-medium text-white">
            {fileName || `Drop your ${destination || 'travel'} itinerary`}
          </p>
          <p className="text-sm text-zinc-500 mt-1">PDF, PNG, JPG up to 10MB</p>
        </div>
      </div>
    </div>
  )
}

2. ProcessingAnimation.tsx — real-time checks

'use client'
import { useEffect, useState } from 'react'

interface Check {
  id: string
  label: string
  status: 'pending' | 'running' | 'done'
}

const CHECKS: Omit<Check, 'status'>[] = [
  { id: 'visa', label: 'Checking visa requirements' },
  { id: 'timing', label: 'Validating layover times' },
  { id: 'insurance', label: 'Verifying travel insurance' },
  { id: 'transfers', label: 'Checking airport transfers' },
  { id: 'hotels', label: 'Validating check-in times' },
]

export default function ProcessingAnimation() {
  const [checks, setChecks] = useState<Check[]>(
    CHECKS.map(c => ({ ...c, status: 'pending' }))
  )

  useEffect(() => {
    CHECKS.forEach((_, i) => {
      setTimeout(() => {
        setChecks(prev => prev.map((c, idx) => 
          idx === i ? { ...c, status: 'running' } : c
        ))
        setTimeout(() => {
          setChecks(prev => prev.map((c, idx) =>
            idx === i ? { ...c, status: 'done' } : c
          ))
        }, 800)
      }, i * 600)
    })
  }, [])

  return (
    <div className="space-y-3">
      {checks.map(check => (
        <div key={check.id} className="flex items-center gap-3">
          <div className={`w-5 h-5 rounded-full border-2 flex items-center justify-center
            ${check.status === 'done' ? 'border-emerald-500 bg-emerald-500/20' : 
              check.status === 'running' ? 'border-violet-500 animate-pulse' : 'border-zinc-700'}`}>
            {check.status === 'done' && (
              <svg width="12" height="12" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="3" className="text-emerald-400"><path d="M20 6L9 17l-5-5"/></svg>
            )}
          </div>
          <span className={`text-sm ${check.status === 'pending' ? 'text-zinc-600' : 'text-zinc-300'}`}>
            {check.label}{check.status === 'running' && '...'}
          </span>
        </div>
      ))}
    </div>
  )
}

3. ScoreCard.tsx — circular progress 7.2/10

interface ScoreCardProps {
  score: number
  maxScore?: number
  breakdown: { category: string; score: number; max: number }[]
}

export default function ScoreCard({ score, maxScore = 10, breakdown }: ScoreCardProps) {
  const percentage = (score / maxScore) * 100
  const circumference = 2 * Math.PI * 52
  const offset = circumference - (percentage / 100) * circumference

  const getColor = () => {
    if (percentage >= 80) return '#10b981'
    if (percentage >= 60) return '#f59e0b'
    return '#ef4444'
  }

  return (
    <div className="bg-zinc-900 border border-zinc-800 rounded-2xl p-6">
      <div className="flex items-center gap-6">
        <div className="relative w-28 h-28">
          <svg className="transform -rotate-90 w-28 h-28">
            <circle cx="56" cy="56" r="52" stroke="#27272a" strokeWidth="8" fill="none" />
            <circle
              cx="56" cy="56" r="52"
              stroke={getColor()}
              strokeWidth="8"
              fill="none"
              strokeLinecap="round"
              strokeDasharray={circumference}
              strokeDashoffset={offset}
              className="transition-all duration-1000"
            />
          </svg>
          <div className="absolute inset-0 flex flex-col items-center justify-center">
            <span className="text-3xl font-bold text-white">{score}</span>
            <span className="text-xs text-zinc-500">/ {maxScore}</span>
          </div>
        </div>
        <div className="flex-1 space-y-2">
          {breakdown.map(item => (
            <div key={item.category} className="flex justify-between text-sm">
              <span className="text-zinc-400">{item.category}</span>
              <span className="text-white font-medium">{item.score}/{item.max}</span>
            </div>
          ))}
        </div>
      </div>
    </div>
  )
}

4. IssuesList.tsx — severity grouped

'use client'
import { useState } from 'react'

interface Issue {
  id: string
  severity: 'critical' | 'warning' | 'info'
  title: string
  description: string
  fix: string
}

interface IssuesListProps {
  issues: Issue[]
}

const SEVERITY = {
  critical: { label: 'Critical', color: 'text-red-400', bg: 'bg-red-500/10', border: 'border-red-500/20' },
  warning: { label: 'Warning', color: 'text-amber-400', bg: 'bg-amber-500/10', border: 'border-amber-500/20' },
  info: { label: 'Tip', color: 'text-blue-400', bg: 'bg-blue-500/10', border: 'border-blue-500/20' },
}

export default function IssuesList({ issues }: IssuesListProps) {
  const [openId, setOpenId] = useState<string | null>(issues[0]?.id)

  const grouped = issues.reduce((acc, issue) => {
    (acc[issue.severity] ||= []).push(issue)
    return acc
  }, {} as Record)

  return (
    <div className="space-y-6">
      {(['critical', 'warning', 'info'] as const).map(sev => {
        const items = grouped[sev]; if (!items?.length) return null
        const cfg = SEVERITY[sev]
        return (
          <div key={sev}>
            <h4 className={`text-sm font-medium mb-3 ${cfg.color}`}>
              {cfg.label} ({items.length})
            </h4>
            <div className="space-y-2">
              {items.map(issue => (
                <div key={issue.id} className={`border rounded-xl ${cfg.border} ${cfg.bg}`}>
                  <button onClick={() => setOpenId(openId === issue.id ? null : issue.id)}
                    className="w-full px-4 py-3 flex justify-between items-center text-left">
                    <span className="font-medium text-white text-sm">{issue.title}</span>
                    <svg className={`w-4 h-4 transition-transform ${openId === issue.id ? 'rotate-180' : ''}`} fill="none" stroke="currentColor" viewBox="0 0 24 24"><path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M19 9l-7 7-7-7"/></svg>
                  </button>
                  {openId === issue.id && (
                    <div className="px-4 pb-4 text-sm space-y-2">
                      <p className="text-zinc-300">{issue.description}</p>
                      <p className="text-zinc-400"><span className="text-emerald-400">Fix:</span> {issue.fix}</p>
                    </div>
                  )}
                </div>
              ))}
            </div>
          </div>
        )
      })}
    </div>
  )
}

5. EmailCapture.tsx — gated full report

'use client'
import { useState } from 'react'

interface EmailCaptureProps {
  reportId: string
  onSuccess?: () => void
}

export default function EmailCapture({ reportId, onSuccess }: EmailCaptureProps) {
  const [email, setEmail] = useState('')
  const [loading, setLoading] = useState(false)

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault()
    setLoading(true)
    await fetch('/api/report', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify({ email, reportId })
    })
    setLoading(false)
    onSuccess?.()
    // Track conversion
    window.posthog?.capture('email_captured', { reportId })
  }

  return (
    <div className="bg-gradient-to-b from-violet-950/40 to-zinc-950 border border-violet-900/50 rounded-2xl p-6">
      <h3 className="text-lg font-semibold text-white mb-1">Get your full report</h3>
      <p className="text-sm text-zinc-400 mb-4">Detailed fixes, visa links, and printable checklist</p>
      <form onSubmit={handleSubmit} className="flex gap-2">
        <input
          type="email"
          required
          value={email}
          onChange={e => setEmail(e.target.value)}
          placeholder="you@email.com"
          className="flex-1 px-4 py-2.5 bg-zinc-900 border border-zinc-800 rounded-lg text-sm text-white placeholder-zinc-600 focus:outline-none focus:border-violet-500"
        />
        <button
          type="submit"
          disabled={loading}
          className="px-5 py-2.5 bg-white text-black rounded-lg text-sm font-medium hover:bg-zinc-200 disabled:opacity-50"
        >
          {loading ? 'Sending...' : 'Send report'}
        </button>
      </form>
      <p className="text-[11px] text-zinc-600 mt-3">No spam. Unsubscribe anytime.</p>
    </div>
  )
}
