4 NB02 Implementation

nb02_lite.py

Simplified 15-rule engine 
class NB02Lite:
    """Simplified NB02 for GTM tool - 15 rules vs 127 in production"""
    
    CRITICAL_WEIGHT = 2.0
    WARNING_WEIGHT = 1.0
    INFO_WEIGHT = 0.3
    
    def __init__(self, itinerary_data):
        self.data = itinerary_data
        self.issues = []
    
    def check_connections(self):
        """CONN_001, CONN_002"""
        for flight in self.data.get('flights', []):
            if flight.get('layover_mins'):
                if flight['type'] == 'international' and flight['layover_mins'] < 180:
                    self.issues.append({
                        'id': 'CONN_001',
                        'severity': 'critical',
                        'message': f"International connection in {flight['airport']} only {flight['layover_mins']}m",
                        'fix': 'Minimum 3 hours recommended for immigration + security'
                    })
                elif flight['type'] == 'domestic' and flight['layover_mins'] < 60:
                    self.issues.append({
                        'id': 'CONN_002',
                        'severity': 'critical',
                        'message': f"Domestic connection {flight['layover_mins']}m",
                        'fix': 'Minimum 60 minutes for domestic'
                    })
    
    def check_transfers(self):
        """TRF_001, TRF_002"""
        for hotel in self.data.get('hotels', []):
            if hotel['distance_from_airport_km'] > 20 and not hotel.get('transfer_included'):
                self.issues.append({
                    'id': 'TRF_001',
                    'severity': 'critical',
                    'message': f"{hotel['name']} is {hotel['distance_from_airport_km']}km from airport, no transfer",
                    'fix': 'Add airport transfer (₹2,000-5,000)'
                })
    
    def check_visas(self):
        """VISA_001"""
        passport = self.data.get('passport_country', 'IN')
        for country in self.data.get('countries_visited', []):
            if requires_visa(passport, country) and not self.data.get('visa_mentioned'):
                self.issues.append({
                    'id': 'VISA_001',
                    'severity': 'critical',
                    'message': f"Visa required for {country} not mentioned",
                    'fix': 'Confirm visa requirements and processing time'
                })
    
    def check_hotels(self):
        """HOT_001, HOT_002, CHK_001"""
        # Distance from city center
        for hotel in self.data.get('hotels', []):
            if hotel.get('distance_from_center_km', 0) > 10:
                self.issues.append({
                    'id': 'HOT_001',
                    'severity': 'warning',
                    'message': f"{hotel['name']} {hotel['distance_from_center_km']}km from center",
                    'fix': 'Consider central location to save commute time'
                })
        
        # Too many changes
        if len(self.data.get('hotels', [])) > self.data.get('nights', 0) / 3:
            self.issues.append({
                'id': 'HOT_002',
                'severity': 'warning',
                'message': f"{len(self.data['hotels'])} hotels in {self.data['nights']} nights",
                'fix': 'Max 1 hotel change per 3 nights recommended'
            })
    
    def calculate_score(self):
        """10 - (critical×2 + warning×1 + info×0.3)"""
        crit = len([i for i in self.issues if i['severity'] == 'critical'])
        warn = len([i for i in self.issues if i['severity'] == 'warning'])
        info = len([i for i in self.issues if i['severity'] == 'info'])
        
        score = 10 - (crit * self.CRITICAL_WEIGHT + 
                     warn * self.WARNING_WEIGHT + 
                     info * self.INFO_WEIGHT)
        return max(0, round(score, 1))
    
    def run_all_checks(self):
        self.check_connections()
        self.check_transfers()
        self.check_visas()
        self.check_hotels()
        # ... 11 more checks
        return {
            'score': self.calculate_score(),
            'issues': self.issues,
            'summary': f"{len(self.issues)} issues found"
        }

Scoring Formula: 10 - (critical × 2 + warning × 1 + info × 0.3)

Example: 1 critical + 3 warnings = 10 - (2 + 3) = 5.0/10
