### Retention Value Calculation

```typescript
// analytics/retention_value.ts
interface RetentionValueInputs {
  annualTripVolume: number;
  avgTripValue: number;
  baselineChurnRate: number; // e.g., 0.25 for 25%
  portalChurnRate: number;
  baselineRepeatRate: number;
  portalRepeatRate: number;
  customerLifetimeYears: number;
}

export function calculateRetentionValue(inputs: RetentionValueInputs) {
  const {
    annualTripVolume,
    avgTripValue,
    baselineChurnRate,
    portalChurnRate,
    baselineRepeatRate,
    portalRepeatRate,
    customerLifetimeYears,
  } = inputs;

  // 1. Churn Reduction Value
  const baselineCustomers = annualTripVolume;
  const portalCustomers = annualTripVolume / (1 - portalChurnRate); // Fewer lost
  const retainedCustomers = portalCustomers - baselineCustomers;
  const churnReductionValue = retainedCustomers * avgTripValue * customerLifetimeYears;

  // 2. Repeat Booking Increase
  const baselineRepeatTrips = annualTripVolume * baselineRepeatRate;
  const portalRepeatTrips = annualTripVolume * portalRepeatRate;
  const additionalRepeatTrips = portalRepeatTrips - baselineRepeatTrips;
  const repeatBookingValue = additionalRepeatTrips * avgTripValue;

  // 3. Referral Value
  const baselineReferralRate = 0.15;
  const portalReferralRate = 0.25;
  const baselineReferrals = annualTripVolume * baselineReferralRate;
  const portalReferrals = annualTripVolume * portalReferralRate;
  const additionalReferrals = portalReferrals - baselineReferrals;
  const referralValue = additionalReferrals * avgTripValue;

  // Total Annual Retention Value
  const totalAnnualValue = churnReductionValue / customerLifetimeYears +
                          repeatBookingValue +
                          referralValue;

  return {
    churnReductionValue: {
      annual: churnReductionValue / customerLifetimeYears,
      total: churnReductionValue,
    },
    repeatBookingValue: {
      annual: repeatBookingValue,
      additionalTrips: additionalRepeatTrips,
    },
    referralValue: {
      annual: referralValue,
      additionalCustomers: additionalReferrals,
    },
    totalAnnualValue,
    lifetimeValue: totalAnnualValue * customerLifetimeYears,
  };
}

// Example calculation
const exampleValue = calculateRetentionValue({
  annualTripVolume: 1000,          // 1000 trips/year
  avgTripValue: 5000,              // $5,000 average trip
  baselineChurnRate: 0.25,         // 25% churn
  portalChurnRate: 0.12,           // 12% churn
  baselineRepeatRate: 0.35,        // 35% repeat
  portalRepeatRate: 0.55,          // 55% repeat
  customerLifetimeYears: 5,        // 5 year average relationship
});

console.log(exampleValue);
// Output:
// {
//   churnReductionValue: { annual: $130,000, total: $650,000 },
//   repeatBookingValue: { annual: $100,000, additionalTrips: 20 },
//   referralValue: { annual: $50,000, additionalCustomers: 10 },
//   totalAnnualValue: $280,000,
//   lifetimeValue: $1,400,000
// }
```

### Efficiency Calculations

```typescript
// analytics/operational_efficiency.ts
interface EfficiencyInputs {
  numberOfAgents: number;
  hourlyRate: number;
  tripsPerAgentPerYear: number;
  baselineInquiriesPerTrip: number;
  portalInquiriesPerTrip: number;
  minutesPerInquiry: number;
}

export function calculateOperationalEfficiency(inputs: EfficiencyInputs) {
  const {
    numberOfAgents,
    hourlyRate,
    tripsPerAgentPerYear,
    baselineInquiriesPerTrip,
    portalInquiriesPerTrip,
    minutesPerInquiry,
  } = inputs;

  const totalTrips = numberOfAgents * tripsPerAgentPerYear;

  // Baseline inquiry handling cost
  const baselineInquiries = totalTrips * baselineInquiriesPerTrip;
  const baselineHours = (baselineInquiries * minutesPerInquiry) / 60;
  const baselineCost = baselineHours * hourlyRate;

  // Portal inquiry handling cost
  const portalInquiries = totalTrips * portalInquiriesPerTrip;
  const portalHours = (portalInquiries * minutesPerInquiry) / 60;
  const portalCost = portalHours * hourlyRate;

  // Savings
  const hoursSaved = baselineHours - portalHours;
  const costSaved = baselineCost - portalCost;
  const inquiriesReduced = baselineInquiries - portalInquiries;

  // Additional capacity (trips per agent that could be handled)
  const tripCapacityIncrease = (hoursSaved / numberOfAgents) / 40; // Assuming 40 hrs/trip

  return {
    baseline: {
      inquiries: baselineInquiries,
      hours: baselineHours,
      cost: baselineCost,
    },
    withPortal: {
      inquiries: portalInquiries,
      hours: portalHours,
      cost: portalCost,
    },
    savings: {
      inquiriesReduced,
      hoursSaved,
      costSaved,
      percentReduction: (inquiriesReduced / baselineInquiries) * 100,
      additionalTripCapacity: tripCapacityIncrease * numberOfAgents,
    },
  };
}

// Example
const efficiencyExample = calculateOperationalEfficiency({
  numberOfAgents: 10,
  hourlyRate: 50,
  tripsPerAgentPerYear: 100,
  baselineInquiriesPerTrip: 5,
  portalInquiriesPerTrip: 1.5, // 70% reduction
  minutesPerInquiry: 3,
});

console.log(efficiencyExample);
// Output:
// {
//   baseline: { inquiries: 5000, hours: 250, cost: $12,500 }
//   withPortal: { inquiries: 1500, hours: 75, cost: $3,750 }
//   savings: {
//     inquiriesReduced: 3500,
//     hoursSaved: 175,
//     costSaved: $8,750,
//     percentReduction: 70%,
//     additionalTripCapacity: 43.75 trips
//   }
// }
```

  };

    paymentDays: number;
  };
  revenue: {
    conversionRate: number;
    avgOrderValue: number;
    upsellRate: number;
    dso: number;
  };
}

export async function getPortalMetrics(agencyId: string, period: 'month' | 'quarter' | 'year'): Promise<PortalMetrics> {
  // Query various data sources
  const [adoption, engagement, retention, operational, revenue] = await Promise.all([
    getAdoptionMetrics(agencyId, period),
    getEngagementMetrics(agencyId, period),
    getRetentionMetrics(agencyId, period),
    getOperationalMetrics(agencyId, period),
    getRevenueMetrics(agencyId, period),
  ]);

  return {
    adoption,
    engagement,
    retention,
    operational,
    revenue,
  };
}
