## Overview

Customer portal engagement drives retention, repeat bookings, and referrals. This document outlines strategies to maximize adoption and ongoing usage throughout the customer lifecycle.

### Engagement Philosophy


```typescript
// components/onboarding/PortalOnboarding.tsx
export function PortalOnboarding({ customer, onComplete }: {
  customer: Customer;
  onComplete: () => void;
}) {
  const [step, setStep] = useState(0);
  const [skipped, setSkipped] = useState(false);

  const steps = [
    {
      title: 'Welcome to your travel portal! 🌍',
      description: 'Everything you need for your upcoming trips, all in one place.',
      component: WelcomeStep,
      action: 'Get Started',
    },
    {
      title: 'Track your trips',
      description: 'View all your bookings, itineraries, and travel documents.',
      component: TripsStep,
      action: 'See My Trips',
    },
    {
      title: 'Stay connected',
      description: 'Message your agent directly and get quick responses.',
      component: MessagesStep,
      action: 'Try Messaging',
    },
    {
      title: 'Access documents anywhere',
      description: 'Download tickets, confirmations, and travel docs on the go.',
      component: DocumentsStep,
      action: 'View Documents',
    },
    {
      title: 'You\'re all set! ✈️',
      description: 'Start exploring your portal. We\'ll notify you about important updates.',
      component: CompleteStep,
      action: 'Explore Portal',
    },
  ];

  const currentStep = steps[step];
  const progress = ((step + 1) / steps.length) * 100;

  return (
    <Dialog open={!skipped} onOpenChange={setSkipped}>
      <DialogContent className="max-w-md">
        {/* Progress bar */}
        <div className="mb-6">
          <div className="h-1 bg-gray-200 rounded-full overflow-hidden">
            <div
              className="h-full bg-blue-500 transition-all duration-300"
              style={{ width: `${progress}%` }}
            />
          </div>
          <p className="text-xs text-gray-500 mt-2 text-center">
            Step {step + 1} of {steps.length}
          </p>
        </div>

        {/* Current step content */}
        <div className="text-center mb-6">
          <h2 className="text-xl font-semibold mb-2">{currentStep.title}</h2>
          <p className="text-gray-600">{currentStep.description}</p>
        </div>

        <currentStep.component customer={customer} />

        {/* Actions */}
        <div className="flex gap-3 mt-6">
          {step > 0 && (
            <Button variant="ghost" onClick={() => setStep(s => s - 1)}>
              Back
            </Button>
          )}
          {step < steps.length - 1 ? (
            <>
              <Button
                variant="ghost"
                className="ml-auto"
                onClick={() => {
                  setSkipped(true);
                  onComplete();
                }}
              >
                Skip
              </Button>
              <Button
                onClick={() => setStep(s => s + 1)}
                className="ml-2"
              >
                {currentStep.action}
              </Button>
            </>
          ) : (
            <Button
              onClick={() => {
                setSkipped(true);
                onComplete();
              }}
              className="ml-auto w-full"
            >
              {currentStep.action}
            </Button>
          )}
        </div>
      </DialogContent>
    </Dialog>
  );
}

// Feature discovery tooltips
export function FeatureTooltips() {
  const [seenTooltips, setSeenTooltips] = useState<Set<string>>(new Set());
  const [activeTooltip, setActiveTooltip] = useState<string | null>(null);

  const tooltips = [
    {
      id: 'trip-countdown',
      selector: '[data-countdown]',
      title: 'Trip Countdown',
      description: 'See exactly how many days until your next adventure!',
      position: 'bottom',
      showAfter: 3, // days since first login
    },
    {
      id: 'document-checklist',
      selector: '[data-document-checklist]',
      title: 'Document Checklist',
      description: 'Track which documents you need and which are complete.',
      position: 'right',
      showAfter: 5,
    },
    {
      id: 'agent-message',
      selector: '[data-agent-chat]',
      title: 'Message Your Agent',
      description: 'Have questions? Send a message directly to your agent.',
      position: 'left',
      showAfter: 7,
    },
    {
      id: 'offline-access',
      selector: '[data-pwa-install]',
      title: 'Install App',
      description: 'Add to home screen for offline access to your documents.',
      position: 'bottom',
      showAfter: 14,
    },
  ];

  useEffect(() => {
    // Determine which tooltip to show based on user tenure and past views
    const daysSinceSignup = getDaysSinceSignup();
    const available = tooltips.filter(
      t => daysSinceSignup >= t.showAfter && !seenTooltips.has(t.id)
    );

    if (available.length > 0) {
      setActiveTooltip(available[0].id);
    }
  }, [seenTooltips]);

  if (!activeTooltip) return null;

  const tooltip = tooltips.find(t => t.id === activeTooltip);
  if (!tooltip) return null;

  return (
    <TooltipProvider>
      <Tooltip open>
```

```typescript
// Notification grouping
export function groupNotifications(notifications: Notification[]): Notification[] {
  const groups: Record<string, Notification[]> = {};

  for (const notification of notifications) {
    const strategy = notificationStrategies[notification.type];
    if (!strategy?.allowGrouping) {
      // Don't group
      groups[notification.id] = [notification];
      continue;
    }

    // Create group key
    const groupKey = `${notification.type}_${notification.tripId}_${notification.customerId}`;
    groups[groupKey] = groups[groupKey] || [];
    groups[groupKey].push(notification);
  }

  // Convert groups to summary notifications
  return Object.values(groups).map(group => {
    if (group.length === 1) return group[0];

    // Create summary
    return {
      ...group[0],
      id: `group_${group[0].type}_${Date.now()}`,
      title: `${group.length} ${group[0].type.replace(/_/g, ' ')}`,
      body: group.map(n => n.title).join(', '),
      data: { grouped: true, count: group.length },
    };
  });
}
```

```typescript
export class PointsEngine {
  async awardPoints(customerId: string, action: string, metadata?: Record<string, unknown>): Promise<number> {
    const actionConfig = engagementActions.find(a => a.type === action);
    if (!actionConfig) return 0;

    // Check daily limit
    if (actionConfig.dailyLimit) {
      const today = new Date().toISOString().split('T')[0];
      const todayCount = await db.query(`
        SELECT COUNT(*) as count
        FROM customer_points
        WHERE customer_id = $1 AND action_type = $2 AND DATE(created_at) = $3
      `, [customerId, action, today]);

      if (todayCount[0].count >= actionConfig.dailyLimit) {
        return 0; // Limit reached
      }
    }

    // Award points
    await db.query(`
      INSERT INTO customer_points (customer_id, action_type, points, metadata, created_at)
      VALUES ($1, $2, $3, $4, NOW())
    `, [customerId, action, actionConfig.points, JSON.stringify(metadata)]);

    // Update customer total
    await db.query(`
      INSERT INTO customer_points_balance (customer_id, balance, updated_at)
      VALUES ($1, $2, NOW())
      ON CONFLICT (customer_id) DO UPDATE SET
        balance = customer_points_balance.balance + $2,
        updated_at = NOW()
    `, [customerId, actionConfig.points]);

    // Check for tier upgrade
    await this.checkTierUpgrade(customerId);

    return actionConfig.points;
  }

  async getBalance(customerId: string): Promise<{ balance: number; tier: string }> {
    const result = await db.query(`
      SELECT balance, tier
      FROM customer_points_balance
      WHERE customer_id = $1
    `, [customerId]);

    return result[0] || { balance: 0, tier: 'Explorer' };
  }

    const { balance } = await this.getBalance(customerId);
    const newTier = loyaltyTiers
      .slice()
      .reverse()
      .find(t => balance >= t.minPoints);

    if (newTier) {
      await db.query(`
        UPDATE customer_points_balance
        SET tier = $1
        WHERE customer_id = $2
      `, [newTier.name, customerId]);

      // Send congratulatory notification
      if (newTier.name !== 'Explorer') {
        await this.notificationService.send(customerId, {
          type: 'tier_upgrade',
          tier: newTier.name,
          benefits: newTier.benefits,
        });
      }
    }
```

```typescript
  async checkAchievements(customerId: string): Promise<string[]> {
    const customer = await this.getCustomerWithStats(customerId);
    const newBadges: string[] = [];

    for (const badge of achievementBadges) {
      // Skip if already earned
      const hasBadge = await this.hasBadge(customerId, badge.id);
      if (hasBadge) continue;

      // Check if requirements met
      if (await this.meetsRequirement(customer, badge.requirement)) {
        await this.awardBadge(customerId, badge.id);
        newBadges.push(badge.id);

      case 'login':
        return customer.loginCount >= requirement.count;
      case 'profile_completion':
        return customer.profileCompletion >= requirement.percentage;
      case 'download_document':
        return customer.documentDownloads >= requirement.count;
      case 'booking_window':
        return customer.bookingWindows.some(w => w >= requirement.days);
      case 'unique_countries':
        return customer.uniqueCountries.size >= requirement.count;
      case 'send_message':
        return customer.messagesSent >= requirement.count;
      case 'trips_booked':
        return customer.tripsBooked >= requirement.count;
      case 'write_review':
        return customer.reviewsWritten >= requirement.count;
      case 'successful_referral':
        return customer.successfulReferrals >= requirement.count;
      case 'login_streak':
```

```typescript
      INSERT INTO re_engagement_log (customer_id, campaign_name, step, sent_at)
      VALUES ($1, $2, $3, NOW())
      ON CONFLICT (customer_id, campaign_name) DO UPDATE SET
        step = EXCLUDED.step,
        sent_at = NOW()
    `, [customerId, campaign, step]);
  }

  async trackResponse(customerId: string, campaign: string, action: 'opened' | 'clicked' | 'converted'): Promise<void> {
    await db.query(`
      INSERT INTO re_engagement_responses (customer_id, campaign_name, action, responded_at)
      VALUES ($1, $2, $3, NOW())
    `, [customerId, campaign, action]);
  }

  async getEffectiveness(campaign: string): Promise<CampaignMetrics> {
    const result = await db.query(`
      WITH campaign_customers AS (
        SELECT DISTINCT customer_id
        FROM re_engagement_log
        WHERE campaign_name = $1
      ),
      sent AS (
        SELECT COUNT(DISTINCT customer_id) as count
        FROM campaign_customers
      ),
      opened AS (
        SELECT COUNT(DISTINCT r.customer_id) as count
        FROM re_engagement_responses r
        JOIN campaign_customers c ON c.customer_id = r.customer_id
        WHERE r.action = 'opened' AND r.campaign_name = $1
      ),
      clicked AS (
        SELECT COUNT(DISTINCT r.customer_id) as count
        FROM re_engagement_responses r
        JOIN campaign_customers c ON c.customer_id = r.customer_id
        WHERE r.action = 'clicked' AND r.campaign_name = $1
      ),
      converted AS (
        SELECT COUNT(DISTINCT r.customer_id) as count
        FROM re_engagement_responses r
        JOIN campaign_customers c ON c.customer_id = r.customer_id
        WHERE r.action = 'converted' AND r.campaign_name = $1
      ),
      reactivated AS (
        SELECT COUNT(DISTINCT c.customer_id) as count
        FROM campaign_customers c
        JOIN customers cu ON cu.id = c.customer_id
        WHERE cu.last_login > (
          SELECT MAX(sent_at) FROM re_engagement_log WHERE customer_id = c.customer_id
        )
      )
      SELECT
        sent.count as sent,
        opened.count as opened,
        clicked.count as clicked,
        converted.count as converted,
        reactivated.count as reactivated,
        ROUND(100.0 * opened.count / NULLIF(sent.count, 0), 1) as open_rate,
        ROUND(100.0 * clicked.count / NULLIF(opened.count, 0), 1) as click_rate,
        ROUND(100.0 * reactivated.count / NULLIF(sent.count, 0), 1) as reactivation_rate
      FROM sent, opened, clicked, converted, reactivated
    `, [campaign]);

    return result[0];
  }
}
```

```typescript
// analytics/engagement_tracking.ts
export interface EngagementEvent {
  customerId: string;
  eventType: string;
  properties?: Record<string, unknown>;
  timestamp: Date;
}

export class EngagementAnalytics {
  async track(event: EngagementEvent): Promise<void> {
    // Send to analytics platform (e.g., Mixpanel, Amplitude)
    await this.analyticsProvider.track({
      userId: event.customerId,
      event: event.eventType,
      properties: event.properties,
      timestamp: event.timestamp,
    });

    // Also store locally for custom analysis
    await db.query(`
      INSERT INTO engagement_events (customer_id, event_type, properties, created_at)
      VALUES ($1, $2, $3, NOW())
    `, [event.customerId, event.eventType, JSON.stringify(event.properties)]);
  }

  async getEngagementScore(customerId: string): Promise<EngagementScore> {
    const [login, activity, feature, social] = await Promise.all([
      this.getLoginScore(customerId),
      this.getActivityScore(customerId),
      this.getFeatureScore(customerId),
      this.getSocialScore(customerId),
    ]);

    const overall = (login * 0.3) + (activity * 0.3) + (feature * 0.25) + (social * 0.15);

    return {
      overall,
      login,
      activity,
      feature,
      social,
      tier: this.getTierFromScore(overall),
    };
  }

  private async getLoginScore(customerId: string): Promise<number> {
    // Based on login frequency and streak
    const result = await db.query(`
      WITH last_30_days AS (
        SELECT COUNT(DISTINCT DATE(created_at)) as days
        FROM engagement_events
        WHERE customer_id = $1
          AND event_type = 'login'
          AND created_at > NOW() - INTERVAL '30 days'
      ),
      streak AS (
        SELECT COUNT(*) as current_streak
        FROM (
          SELECT DATE(created_at), ROW_NUMBER() OVER (ORDER BY created_at DESC) - ROW_NUMBER() OVER (PARTITION BY DATE(created_at) ORDER BY created_at DESC) as grp
          FROM engagement_events
          WHERE customer_id = $1 AND event_type = 'login'
          ORDER BY created_at DESC
          LIMIT 100
        ) t
        GROUP BY grp ORDER BY grp LIMIT 1
      )
      SELECT
        COALESCE(LEAST(days * 3.33, 100), 0) as frequency_score, -- 30 days = 100
        COALESCE(LEAST(current_streak * 14.28, 100), 0) as streak_score -- 7 day streak = 100
      FROM last_30_days, streak
    `, [customerId]);

    return (result[0].frequency_score + result[0].streak_score) / 2;
  }

  private async getActivityScore(customerId: string): Promise<number> {
    const result = await db.query(`
      WITH actions AS (
        SELECT
          SUM(CASE WHEN event_type = 'view_trip' THEN 1 ELSE 0 END) as views,
          SUM(CASE WHEN event_type = 'download_document' THEN 1 ELSE 0 END) as downloads,
          SUM(CASE WHEN event_type = 'send_message' THEN 1 ELSE 0 END) as messages,
          SUM(CASE WHEN event_type = 'make_payment' THEN 1 ELSE 0 END) as payments
        FROM engagement_events
        WHERE customer_id = $1 AND created_at > NOW() - INTERVAL '30 days'
      )
      SELECT
        LEAST(views * 5, 40) +
        LEAST(downloads * 10, 25) +
        LEAST(messages * 5, 25) +
        LEAST(payments * 20, 10) as score
      FROM actions
    `, [customerId]);

    return result[0].score || 0;
  }

  private getTierFromScore(score: number): 'bronze' | 'silver' | 'gold' | 'platinum' {
    if (score >= 80) return 'platinum';
    if (score >= 60) return 'gold';
    if (score >= 40) return 'silver';
    return 'bronze';
  }
}
```
