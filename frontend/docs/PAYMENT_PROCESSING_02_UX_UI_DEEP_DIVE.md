```typescript
/**
 * Payment link creation component
 */
interface PaymentLinkCreateProps {
  tripId?: string;
  invoiceId?: string;
  defaultAmount?: number;
  onSuccess: (link: PaymentLink) => void;
  onCancel: () => void;
}

const PaymentLinkCreate: React.FC<PaymentLinkCreateProps> = ({
  tripId,
  invoiceId,
  defaultAmount,
  onSuccess,
  onCancel
}) => {
  const [amount, setAmount] = useState(defaultAmount || 0);
  const [type, setType] = useState<'FULL' | 'PARTIAL' | 'SPLIT'>('FULL');
  const [splits, setSplits] = useState<Split[]>([]);
  const [expireIn, setExpireIn] = useState(24);
  const [notify, setNotify] = useState(true);
  const [customerEmail, setCustomerEmail] = useState('');
  const [customerPhone, setCustomerPhone] = useState('');
  const [loading, setLoading] = useState(false);
  const [error, setError] = useState<string | null>(null);

  const handleCreate = async () => {
    setError(null);
    setLoading(true);

    try {
      if (type === 'SPLIT') {
        const links = await createSplitLinks({
          tripId,
          invoiceId,
          splits,
          expireIn,
          notify,
          customerEmail,
          customerPhone
        });
        onSuccess(links[0]); // Return first link
      } else {
        const link = await createPaymentLink({
          amount,
          type,
          tripId,
          invoiceId,
          expireIn,
          notify,
          customerEmail,
          customerPhone
        });
        onSuccess(link);
      }
    } catch (err) {
      setError(err instanceof Error ? err.message : 'Failed to create payment link');
    } finally {
      setLoading(false);
    }
  };

  const isFormValid = amount > 0 && (
    type !== 'SPLIT' ||
    splits.length > 0 && splits.every(s => s.amount > 0)
  );

  return (
    <Modal onClose={onCancel}>
      <ModalHeader>
        <h2>Create Payment Link</h2>
      </ModalHeader>

      <ModalBody>
        {/* Trip/Invoice Summary */}
        {(tripId || invoiceId) && (
          <PaymentSummary tripId={tripId} invoiceId={invoiceId} />
        )}

        {/* Amount Input */}
        <AmountInput
          value={amount}
          onChange={setAmount}
          type={type}
          onTypeChange={setType}
        />

        {/* Split Configuration */}
        {type === 'SPLIT' && (
          <SplitConfiguration
            splits={splits}
            onChange={setSplits}
            totalAmount={amount}
          />
        )}

        {/* Options */}
        <PaymentOptions
          expireIn={expireIn}
          onExpireInChange={setExpireIn}
          notify={notify}
          onNotifyChange={setNotify}
        />

        {/* Customer Details */}
        {notify && (
          <CustomerDetails
            email={customerEmail}
            onEmailChange={setCustomerEmail}
            phone={customerPhone}
            onPhoneChange={setCustomerPhone}
          />
        )}

        {/* Error Display */}
        {error && <ErrorMessage message={error} />}
      </ModalBody>

      <ModalFooter>
        <Button variant="secondary" onClick={onCancel}>
          Cancel
        </Button>
        <Button
          variant="primary"
          onClick={handleCreate}
          disabled={!isFormValid || loading}
          loading={loading}
        >
          Create Link
        </Button>
      </ModalFooter>
    </Modal>
  );
};
```

        {!imageLoaded && (
          <div
            className="bg-gray-100 animate-pulse"
            style={{ width: size, height: size }}
          />
        )}
        {paymentLink.qrCode && (
          <img
            src={paymentLink.qrCode}
            alt="Payment QR Code"
            width={size}
            height={size}
            className={`rounded-lg ${imageLoaded ? 'block' : 'hidden'}`}
            onLoad={() => setImageLoaded(true)}
          />
        )}
      </div>

      <div className="text-center">
        <p className="text-sm text-gray-600">
          Scan with any UPI app to pay
        </p>
        <p className="text-xs text-gray-500 mt-1">
          {formatCurrency(paymentLink.amount, paymentLink.currency)}
        </p>
      </div>

      {downloadable && (
        <Button
          variant="outline"
          size="sm"
          onClick={handleDownload}
          icon="download"
        >
          Download QR Code
        </Button>
      )}
    </div>
  );
};

/**
 * Payment timeline component
 */
interface PaymentTimelineProps {
  transaction: Transaction;
  live?: boolean;
}

const PaymentTimeline: React.FC<PaymentTimelineProps> = ({
  transaction,
  live = false
}) => {
  const [currentStatus, setCurrentStatus] = useState(transaction.status);

  useEffect(() => {
    if (!live) return;

    // Subscribe to live updates
    const unsubscribe = subscribeToPaymentUpdates(
      transaction.id,
      (update) => {
        setCurrentStatus(update.status);
      }
    );

    return unsubscribe;
  }, [transaction.id, live]);

  const events = useMemo(() => {
    const baseEvents = [
      {
        key: 'created',
        label: 'Link Created',
        timestamp: transaction.createdAt,
        status: 'complete'
      },
      {
        key: 'paid',
        label: 'Payment Received',
        timestamp: transaction.paidAt,
        status: transaction.paidAt ? 'complete' : currentStatus === 'FAILED' ? 'error' : 'pending'
      },
      {
        key: 'confirmed',
        label: 'Booking Confirmed',
        timestamp: transaction.paidAt,
        status: currentStatus === 'COMPLETED' ? 'complete' : 'pending'
      }
    ];

    return baseEvents;
  }, [transaction, currentStatus]);

  return (
    <div className="payment-timeline">
      {events.map((event, index) => (
        <TimelineEvent
          key={event.key}
          event={event}
          isLast={index === events.length - 1}
        />
      ))}
    </div>
  );
};

const TimelineEvent: React.FC<{
  event: TimelineEventData;
  isLast: boolean;
}> = ({ event, isLast }) => {
  const getStatusIcon = () => {
    switch (event.status) {
      case 'complete':
        return <CheckCircle className="w-5 h-5 text-green-500" />;
      case 'error':
        return <XCircle className="w-5 h-5 text-red-500" />;
      case 'pending':
        return <Clock className="w-5 h-5 text-gray-300" />;
      default:
        return <Circle className="w-5 h-5 text-gray-300" />;
    }
  };

  return (
    <div className="flex gap-4">
      <div className="flex flex-col items-center">
        {getStatusIcon()}
        {!isLast && (
          <div className="w-0.5 flex-1 bg-gray-200 mt-1" />
        )}
      </div>

      <div className="flex-1 pb-6">
        <p className="font-medium text-gray-900">{event.label}</p>
        {event.timestamp && (
          <p className="text-sm text-gray-500">
            {formatDateTime(event.timestamp)}
          </p>
        )}
        {event.status === 'pending' && (
          <p className="text-sm text-gray-400 mt-1">
            Waiting for confirmation...
          </p>
        )}
      </div>
    </div>
  );
};
