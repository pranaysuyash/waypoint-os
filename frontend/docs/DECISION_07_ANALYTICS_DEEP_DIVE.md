## 2. Metric Taxonomy

### Primary Metrics (P0)

| Metric | Definition | Target | Why It Matters |
|--------|------------|--------|----------------|
| **Decision Accuracy** | % of decisions that match agent judgment | вүҘ85% | Core effectiveness |
| **Override Rate** | % of decisions agents override | 10-20% | Sweet spot indicates helpful but not overconfident |
| **Time-to-Decision** | Seconds from data-ready to decision | <2s | Agent productivity |
| **Confidence Calibration** | Correlation between confidence and accuracy | вүҘ0.8 | Trust building |

### Secondary Metrics (P1)

| Metric | Definition | Target | Why It Matters |
|--------|------------|--------|----------------|
| **Funnel Conversion** | Inquiry вҶ’ Quote вҶ’ Booking | Varies | Revenue impact |
| **Extraction Coverage** | Fields auto-populated | вүҘ90% | Data entry savings |
| **Risk Detection Rate** | Risks correctly flagged | вүҘ95% | Compliance & protection |
| **False Positive Rate** | Incorrect risk flags | вүӨ5% | Agent frustration |

### Diagnostic Metrics (P2)

| Metric | Definition | Use Case |
|--------|------------|----------|
| **Feature Importance** | Which features drive decisions | Model debugging |
| **State Transition Rate** | How trips move between states | Workflow optimization |
| **Agent Agreement Variance** | How much agents disagree | Training opportunities |
| **Temporal Drift** | Accuracy changes over time | Model retraining triggers |

### Accuracy by Decision Type

```typescript
interface AccuracyMetrics {
  // Overall accuracy
  overallAccuracy: number;           // 0-1

  // By decision type
  byDecisionType: {
    stateClassification: {
      accuracy: number;              // Correct state assignment
      confusionMatrix: ConfusionMatrix;
    };
    riskFlagging: {
      precision: number;             // True positives / (TP + FP)
      recall: number;                // True positives / (TP + FN)
      f1Score: number;
      costPerFalsePositive: number;  // Business cost
      costPerFalseNegative: number;  // Business cost
    };
    completeness: {
      accuracy: number;              // Correct missing field detection
      coverage: number;              // Fields successfully extracted
    };
    recommendation: {
      acceptanceRate: number;        // % of recommendations acted on
      actionRate: number;            // Led to productive action
    };
  };

  // By confidence level
  byConfidenceBand: {
    low: { accuracy: number; count: number };    // 0-40%
    medium: { accuracy: number; count: number }; // 40-70%
    high: { accuracy: number; count: number };   // 70-90%
    veryHigh: { accuracy: number; count: number }; // 90-100%
  };

  // Temporal accuracy
  temporalTrend: {
    daily: Array<{ date: string; accuracy: number }>;
    drift: number;                   // Change over time
    seasonality: SeasonalityPattern;
  };
}

interface ConfusionMatrix {
  matrix: number[][];                // [predicted][actual]
  labels: string[];                  // State names
  normalized: number[][];            // Row-normalized
}
```

### Funnel Metrics Framework

```typescript
interface FunnelMetrics {
  // Stage-by-stage conversion
  stages: {
    inquiry: {
      count: number;
      conversionTo: {
        extraction: number;          // % that reach extraction
        decision: number;            // % that reach decision
        quote: number;               // % that get quoted
        booking: number;             // % that book
      };
    };
    extraction: {
      count: number;
      avgDuration: number;           // Time to extract
      fieldCoverage: Record<string, number>; // % filled per field
      conversionTo: {
        decision: number;
        quote: number;
        booking: number;
      };
    };
    decision: {
      count: number;
      avgConfidence: number;
      overrideRate: number;
      conversionTo: {
        quote: number;
        booking: number;
      };
    };
    quote: {
      count: number;
      avgTimeToSend: number;
      conversionTo: {
        booking: number;
      };
    };
    booking: {
      count: number;
      avgValue: number;
      timeToBook: number;
    };
  };

  // Derived metrics
  dropoffPoints: Array<{
    stage: string;
    dropoffRate: number;
    topReasons: string[];
    suggestedActions: string[];
  }>;

  // Time-based funnel
  timeInStage: {
    extraction: { p50: number; p95: number; p99: number };
    decision: { p50: number; p95: number; p99: number };
    quote: { p50: number; p95: number; p99: number };
  };
}
```

interface HistoricalAnalytics {
  // Time series queries
  getTimeSeries(params: {
    metric: MetricType;
    granularity: 'hour' | 'day' | 'week' | 'month';
    startDate: Date;
    endDate: Date;
    segment?: SegmentFilter;
  }): Promise<TimeSeriesData>;

  // Comparative analysis
  comparePeriods(params: {
    metric: MetricType;
    period1: DateRange;
    period2: DateRange;
    segment?: SegmentFilter;
  }): Promise<ComparisonResult>;

  // Cohort analysis
  cohortAnalysis(params: {
    cohortBy: 'agent' | 'tripType' | 'destination' | 'customerSegment';

    startDate: Date;
    endDate: Date;
  }): Promise<CohortData>;

  // Funnel analysis over time
  funnelTrend(params: {
    startDate: Date;
    endDate: Date;
    granularity: 'day' | 'week';
  }): Promise<FunnelTrendData>;
}

interface TimeSeriesData {

    timestamp: Date;
    value: number;
    confidence?: number;             // For statistical confidence

  aggregation: AggregationStats;
  trend: TrendAnalysis;
}

interface TrendAnalysis {
  direction: 'up' | 'down' | 'stable';
  changePercent: number;
  changeAbsolute: number;
  significance: 'significant' | 'not_significant';  // Statistical significance
  seasonality?: SeasonalityPattern;
}

interface ComparisonResult {

  period2: AggregationStats;
  difference: {
    absolute: number;
    percent: number;
    significant: boolean;
  };
  insights: string[];
}

### A/B Test Framework Implementation

```typescript
interface ABTest {
  id: string;
  name: string;
  description: string;
  status: 'draft' | 'running' | 'completed' | 'rolled_back';

  // Test configuration
  config: {
    trafficSplit: number;             // 0-1, proportion for treatment
    minSampleSize: number;            // Minimum for statistical power
    maxDuration: number;              // Maximum days to run
    significanceLevel: number;        // Typically 0.05
    minEffectSize: number;            // Minimum detectable effect
  };

  // Treatment definition
  treatment: {
    type: 'model' | 'threshold' | 'feature' | 'ux' | 'ensemble';
    changes: Record<string, any>;
    rollbackPlan: string;
  };

  // Metrics to track
  primaryMetric: 'accuracy' | 'override_rate' | 'conversion';
  secondaryMetrics: string[];

  // Guardrail metrics (must not degrade)
  guardrails: {
    metric: string;
    maxDegradation: number;           // Max acceptable drop
  }[];

  // Results
  results?: {
    control: MetricValues;
    treatment: MetricValues;
    difference: MetricDifference;
    significance: TestStatistics;
    recommendation: 'rollout' | 'extend' | 'rollback' | 'inconclusive';
  };
}

interface ABTestService {
  // Create and configure test
  createTest(config: ABTest): Promise<string>;

  // Assign variant to trip
  assignVariant(tripId: string, testId: string): 'control' | 'treatment';

  // Record metrics
  recordMetric(testId: string, variant: 'control' | 'treatment',
               metric: string, value: number): Promise<void>;

  // Get current status
  getStatus(testId: string): Promise<ABTestStatus>;

  // Calculate significance
  calculateSignificance(testId: string): Promise<TestStatistics>;

  // Auto-check for early stopping
  checkEarlyStopping(testId: string): Promise<{
    shouldStop: boolean;
    reason?: string;
  }>;
}
```

      this.getExtractionCount(period),
      this.getDecisionCount(period),
      this.getQuoteCount(period),
      this.getBookingCount(period)

    const funnel: FunnelStage[] = [
      { name: 'inquiry', count: stages[0] },
      { name: 'extraction', count: stages[1] },
      { name: 'decision', count: stages[2] },
      { name: 'quote', count: stages[3] },
      { name: 'booking', count: stages[4] }
    ];

    // Calculate conversion rates
    for (let i = 1; i < funnel.length; i++) {
      funnel[i].conversionRate = funnel[i].count / funnel[0].count;
      funnel[i].dropoffRate = 1 - (funnel[i].count / funnel[i - 1].count);
    }

    // Identify dropoff points
    const dropoffPoints = funnel
      .filter(s => s.dropoffRate && s.dropoffRate > 0.1)
      .map(s => ({
        stage: s.name,
        dropoffRate: s.dropoffRate!,
        suggestedActions: this.getDropoffActions(s.name)
      }));

    return { funnel, dropoffPoints };
  }

  async getAgentLeaderboard(period: DateRange, limit: number = 10):

// Alerting Service
class AlertingService {
  private rules: AlertRule[];
  private channels: Map<string, NotificationChannel>;

  // Process metrics and check for alerts
  async processMetrics(metrics: RealTimeMetrics): Promise<void> {
    const alerts: Alert[] = [];

    for (const rule of this.rules) {
      const result = await this.evaluateRule(rule, metrics);
      if (result.shouldAlert) {
        alerts.push({
          ruleId: rule.id,
          severity: rule.severity,
          message: this.formatAlertMessage(rule, result),
          timestamp: new Date(),
          context: result.context
        });
      }
    }

    // Send alerts
    for (const alert of alerts) {
      await this.sendAlert(alert);
    }
  }

  // Evaluate a single alert rule
  private async evaluateRule(rule: AlertRule, metrics: RealTimeMetrics):
      Promise<AlertEvaluation> {
    const window = this.parseWindow(rule.window);
    const metricValue = this.getMetricForWindow(metrics, rule.condition, window);

    // Compare against threshold
    const exceedsThreshold = this.compare(metricValue, rule.threshold);

    if (!exceedsThreshold) {
      return { shouldAlert: false };
    }

    // Check for cooldown (don't alert too frequently)
    const lastAlert = await this.getLastAlertTime(rule.id);
    const cooldown = this.getCooldown(rule.severity);
    if (Date.now() - lastAlert < cooldown) {
      return { shouldAlert: false };
    }

    return {
      shouldAlert: true,
      context: {
        metric: rule.condition,
        threshold: rule.threshold,
        actual: metricValue,
        window: rule.window
      }
    };
  }

  // Send alert to appropriate channels
  private async sendAlert(alert: Alert): Promise<void> {
    const channels = this.getChannelsForSeverity(alert.severity);

    for (const channel of channels) {
      try {
        await channel.send(alert);
      } catch (error) {
        console.error(`Failed to send alert to ${channel.name}:`, error);
        // Try fallback channel
        await this.sendToFallback(alert, error);
      }
    }

    // Record alert
    await this.recordAlert(alert);
  }

  // Check if alert should be escalated
  async checkEscalation(alertId: string): Promise<void> {
    const alert = await this.getAlert(alertId);

    if (alert.status !== 'open') {
      return;
    }

    const escalationPlan = this.getEscalationPlan(alert.severity);
    const timeSinceAlert = Date.now() - alert.timestamp.getTime();

    for (const step of escalationPlan) {
      if (timeSinceAlert > step.afterMinutes * 60 * 1000 &&
          !alert.escalationSteps.includes(step.level)) {
        await this.escalateAlert(alert, step);
      }
    }
  }
}

interface Alert {
  ruleId: string;
  severity: 'critical' | 'warning' | 'info';
  message: string;
  timestamp: Date;
  context: any;
  status?: 'open' | 'acknowledged' | 'resolved';
  escalationSteps?: string[];
}
